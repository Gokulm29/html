pipeline {
    agent any
    environment {
        DEPLOY_DIR = "/var/www/html" // Change this to your web server's deployment path
    }
    stages {
        stage('Prepare Workspace') {
            steps {
                echo 'Setting up the workspace for deployment...'
                // Optionally clean up workspace to avoid leftover files
                cleanWs()
            }
        }
        stage('Checkout Code') {
            steps {
                echo 'Cloning the repository containing the HTML project...'
                git branch: 'main', url: 'https://github.com/username/html-project.git'
            }
        }
        stage('Build') {
            steps {
                echo 'Performing any necessary build steps...'
                // Static HTML typically doesn't need a build step; add commands if needed
            }
        }
        stage('Test') {
            steps {
                echo 'Running linting and validations...'
                // Example: Add a linter for HTML validation
                // Install htmlhint or similar tools in the environment to enable this
                sh 'htmlhint .'
            }
        }
        stage('Deploy') {
            steps {
                echo 'Deploying HTML files to the web server...'
                sh """
                # Ensure the deployment directory is clean
                sudo rm -rf ${DEPLOY_DIR}/*
                # Copy the entire workspace to the deployment directory
                sudo cp -r * ${DEPLOY_DIR}
                """
            }
        }
    }
    post {
        success {
            echo 'Deployment completed successfully!'
        }
        failure {
            echo 'Pipeline failed. Check the logs for details.'
        }
    }
}
